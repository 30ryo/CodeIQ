import nltk, re, csv
from nltk.corpus import gutenberg

words = [word for word
         in gutenberg.words('austen-sense.txt')
         if re.match(r'[a-zA-Z]+',word)]
fdw = nltk.FreqDist(words)

tokens = fdw.keys()
vals = fdw.values() 
freqs = [fdw.freq(word)*100 for word in fdw.keys()]

with open('Freq.csv', 'w') as f:
    writer = csv.writer(f, lineterminator = "\n")
    writer.writerow(('Token','Frequency','Frequency(%)'))
    for t,v,f in zip(tokens, vals, freqs):
        writer.writerow((t,v,f))
